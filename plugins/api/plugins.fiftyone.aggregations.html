Example

const [aggregate, points, loading] = foa.useAggregation({
  dataset,
  filters,
  view,
});

React.useEffect(() => {
  aggregate(
    [
      new foa.aggregations.Values({
        fieldOrExpr: "id",
      }),
      new foa.aggregations.Values({
        fieldOrExpr: `${path}.point.coordinates`,
      }),
    ],
    dataset.name
  );
}, [dataset, filters, view, path]);
